If you need to use multiple 'if' statements,
you can use one 'switch' statement.

switch ( grade ) {

	case "Regular":

		console.log("$3.15");

		break;



	case "Premium":

		console.log("$3.47");

		break;



	case "Diesel":

		console.log("$3.68");

		break;


	
default:

		console.log("not valid");

}
